```sql
-- AI Preview Jobs table
CREATE TABLE IF NOT EXISTS ai_preview_jobs (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  status        TEXT NOT NULL DEFAULT 'pending'
                  CHECK (status IN ('pending','processing','completed','failed')),
  input_image_url  TEXT,

  style_type       TEXT NOT NULL DEFAULT 'hair'
                     CHECK (style_type IN ('hair','makeup','nails','full_look')),
  result_image_url TEXT,
  error_message    TEXT,
  tokens_used      INTEGER DEFAULT 0,
  model_used       TEXT DEFAULT 'dall-e-3',
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- AI Preview Saved Looks table
CREATE TABLE IF NOT EXISTS ai_preview_looks (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  job_id          UUID NOT NULL REFERENCES ai_preview_jobs(id) ON DELETE CASCADE,
  title           TEXT,
  description     TEXT,
  is_public       BOOLEAN DEFAULT FALSE,
  likes_count     INTEGER DEFAULT 0,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- AI Preview Usage Tracking
CREATE TABLE IF NOT EXISTS ai_preview_usage (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  month_year    TEXT NOT NULL,        -- e.g. '2026-03'
  previews_used INTEGER DEFAULT 0,
  tokens_used   INTEGER DEFAULT 0,
  UNIQUE (user_id, month_year)
);

-- RLS Policies
ALTER TABLE ai_preview_jobs    ENABLE ROW LEVEL SECURITY;
ALTER TABLE ai_preview_looks   ENABLE ROW LEVEL SECURITY;
ALTER TABLE ai_preview_usage   ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can manage their own preview jobs"
  ON ai_preview_jobs FOR ALL
  USING (auth.uid() = user_id);

CREATE POLICY "Users can manage their own looks"
  ON ai_preview_looks FOR ALL
  USING (auth.uid() = user_id);

CREATE POLICY "Public looks are viewable by all"
  ON ai_preview_looks FOR SELECT
  USING (is_public = TRUE);

CREATE POLICY "Users can manage their own usage"
  ON ai_preview_usage FOR ALL
  USING (auth.uid() = user_id);
```

**Edge Function:** `supabase/functions/generate-ai-preview/index.ts`

```typescript
import { serve } from "https://deno.land/std@0.208.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";
import OpenAI from "https://esm.sh/openai@4";

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response("ok", { headers: corsHeaders });
  }

  try {
    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
    );

    const authHeader = req.headers.get("Authorization")!;
    const { data: { user }, error: authError } = await supabase.auth.getUser(
      authHeader.replace("Bearer ", "")
    );
    if (authError || !user) {
      return new Response(JSON.stringify({ error: "Unauthorized" }), {
        status: 401, headers: { ...corsHeaders, "Content-Type": "application/json" }
      });
    }

    const { prompt, style_type, input_image_url } = await req.json();

    // Create job record
    const { data: job, error: jobError } = await supabase
      .from("ai_preview_jobs")
      .insert({ user_id: user.id, prompt, style_type, input_image_url, status: "processing" })
      .select()
      .single();
    if (jobError) throw jobError;

    // Call OpenAI DALL-E 3
    const openai = new OpenAI({ apiKey: Deno.env.get("OPENAI_API_KEY")! });
    const fullPrompt = `Professional beauty salon photo. ${prompt}. Style: ${style_type}. High quality, photorealistic.`;
    const response = await openai.images.generate({
      model: "dall-e-3",
      prompt: fullPrompt,
      n: 1,
      size: "1024x1024",
      quality: "standard",
    });

    const result_image_url = response.data[0]?.url;

    // Update job with result
    await supabase
      .from("ai_preview_jobs")
      .update({ status: "completed", result_image_url, tokens_used: 1, updated_at: new Date().toISOString() })
      .eq("id", job.id);

    // Update usage tracking
    const monthYear = new Date().toISOString().slice(0, 7);
    await supabase.from("ai_preview_usage").upsert(
      { user_id: user.id, month_year: monthYear, previews_used: 1, tokens_used: 1 },
      { onConflict: "user_id,month_year", ignoreDuplicates: false }
    );
      JSON.stringify({ job_id: job.id, result_image_url }),
      { headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({ error: error.message }),
      { status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" } }
    );
  }
});
```

### Data Flow Example (Booking):

```typescript
// 1. User clicks "Prenota" button
onClick(() => navigate('/booking'))

// 2. Frontend validates input
const validateBooking = (data: BookingData) => {
  if (!data.service_id) throw new Error('Seleziona servizio');
  if (!data.booking_date) throw new Error('Seleziona data');
}

// 3. Call Edge Function (server-side)
const { data, error } = await supabase.functions.invoke('create-booking', {
  body: {
    service_id: service.id,
    provider_id: provider.id,
    booking_date: selectedDate,
    booking_time: selectedTime,
    payment_method: 'stripe'
  }
})

// 4. Edge Function validates server-side
// - Check service exists & available
// - Check provider schedule
// - Verify payment method
// - Calculate price (with taxes, fees)
// - Create Stripe payment intent

// 5. Database transaction (atomic)
// BEGIN TRANSACTION;
//   INSERT INTO bookings (...);
//   UPDATE services SET bookings_count = bookings_count + 1;
//   INSERT INTO notifications (...);
//   INSERT INTO audit_logs (...);
// COMMIT;

// 6. Post-transaction hooks
// - Send confirmation email
// - Send WhatsApp notification
// - Create calendar event
// - Update analytics
// - Trigger 24h reminder (pg_cron)

// 7. Real-time update (Supabase Realtime)
supabase
  .channel('bookings')
  .on('INSERT', payload => {
    toast.success('Nuova prenotazione!')
  })

// 8. Return success to frontend
return { booking_id, payment_url }
```

## 🚀 SETUP SVILUPPO

### Prerequisiti:
- Node.js >= 18.0.0
- npm >= 9.0.0
- Git >= 2.40.0
- Supabase CLI >= 1.180.0

### 1. Clone Repository:

```bash
git clone https://github.com/mikimause93/beauty-style-pro.git
cd beauty-style-pro
npm install
npm run check
```

### 2. Environment Setup:

```env
# Supabase
VITE_SUPABASE_URL=https://your-project.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-key

# Stripe
VITE_STRIPE_PUBLISHABLE_KEY=pk_test_xxx

# Google Maps
VITE_GOOGLE_MAPS_API_KEY=AIzaSyXXX

# OpenAI (Stella AI + AI Preview)
VITE_OPENAI_API_KEY=sk-xxx

# App Config
VITE_APP_NAME="Beauty Style Pro"
VITE_APP_VERSION="4.0.0"
VITE_APP_ENV="development"
```

### 3. Supabase Local Setup:

```bash
supabase login
supabase init
supabase link --project-ref your-project-ref
supabase db pull
supabase start
supabase status
```

### 4. Database Migrations:

```bash
supabase db push          # Apply all pending migrations
# or
supabase migration up     # Apply next migration only
```

### 5. Run Development Server:

```bash
npm run dev               # http://localhost:5173
```

### 6. Run Tests:

```bash
npm test                  # Unit tests (Vitest)
npm run test:ci           # CI mode with coverage
npm run e2e               # End-to-end tests (Playwright)
npm run coverage          # Coverage report
```
